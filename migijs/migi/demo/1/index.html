class MyComponent extends migi.Component {
  constructor(...data) {
    super(...data);
  this.text = 'like';
  }
  get txt() {
    return this.text;
  }
  set txt(v) {
    this.text = v;
  }
  handleClick() {
    this.txt = this.txt == 'like' ? 'unlike' : 'like';
  }
  render() {
    return (
      <p onClick={this.handleClick}>
        You {this.txt} this. Click to toggle.
      </p>
    );
  }
}

migi.render(
  <MyComponent />,
  '#example'
);
